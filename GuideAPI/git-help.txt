--- ПЕРЕД ПОЧАТКОМ РОБОТИ ---
git checkout master        # Перейти на головну гілку
git pull origin master     # Оновити код до найсвіжішої версії

--- РОБОТА В ГІЛЦІ (ФІЧА) ---
git checkout -b feature/назва  # Створити нову гілку і перейти в неї
git status                     # Перевірити, які файли змінено
git add .                      # Додати всі зміни до коміту
git commit -m "Опис змін"      # Зафіксувати зміни локально
git push -u origin feature/назва # Відправити гілку на GitHub

--- ПЕРЕМИКАННЯ МІЖ ГІЛКАМИ ---
git branch                     # Подивитися список локальних гілок
git checkout назва-гілки       # Перейти на вже існуючу гілку

--- ЯКЩО ПРИЛЕТІЛИ ОНОВЛЕННЯ В MASTER ---
(будучи у своїй гілці)
git merge master               # Підтягнути зміни з master у свою гілку
